GUISkin.box 盒子

var box : GUIStyle

Description描述

Style used by default for GUI.Box controls.

用于 GUI.Box控件的默认样式。

  • C#
  • JavaScript
using UnityEngine;
using System.Collections;

public class example : MonoBehaviour {
	public GUIStyle style;
	void OnGUI() {
		GUI.skin.box = style;
		GUILayout.Box("This is a box.");
	}
}
// Modifies only the box style of the current GUISkin
//修改当前GUISkin的box样式
var style : GUIStyle;

function OnGUI() {
	GUI.skin.box = style;
	GUILayout.Box("This is a box.");
}
