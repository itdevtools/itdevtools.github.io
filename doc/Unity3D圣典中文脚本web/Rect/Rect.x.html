Rect.x 左坐标

var x : float

Description描述

Left coordinate of the rectangle.

矩形的左坐标。

  • C#
  • JavaScript
using UnityEngine;
using System.Collections;

public class example : MonoBehaviour {
	public Rect rect = new Rect(10, 10, 100, 100);
	public void Awake() {
		print(rect.x);
		rect.x = 20;
	}
}
var rect = Rect (10, 10, 100, 100);
print(rect.x);
rect.x = 20;
