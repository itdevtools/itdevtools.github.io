Mathf.CeilToInt 最小整数

static function CeilToInt (f : float) : int

Description描述

Returns the smallest integer greater to or equal to f.

返回最小的整数大于或等于f。

  • C#
  • JavaScript
using UnityEngine;
using System.Collections;

public class example : MonoBehaviour {
	public void Awake() {
		Debug.Log(Mathf.CeilToInt(10.0F));
		Debug.Log(Mathf.CeilToInt(10.2F));
		Debug.Log(Mathf.CeilToInt(10.7F));
		Debug.Log(Mathf.CeilToInt(-10.0F));
		Debug.Log(Mathf.CeilToInt(-10.2F));
		Debug.Log(Mathf.CeilToInt(-10.7F));
	}
}
// Prints 10
Debug.Log(Mathf.CeilToInt(10.0));
// Prints 11
Debug.Log(Mathf.CeilToInt(10.2));
// Prints 11
Debug.Log(Mathf.CeilToInt(10.7));
// Prints -10
Debug.Log(Mathf.CeilToInt(-10.0));
// Prints -10
Debug.Log(Mathf.CeilToInt(-10.2));
// Prints -10
Debug.Log(Mathf.CeilToInt(-10.7));
